> cp diva-distribution/bin/Diva.OpenSimServices.dll
> opensim-0.7.1-pre-release/bin/
> cp diva-distribution/bin/Diva.Modules.dll opensim-0.7.1-pre-release/bin/
> cp -pr diva-distribution/WifiPages opensim-0.7.1-pre-release
