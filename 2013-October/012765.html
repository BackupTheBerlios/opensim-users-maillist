It will automatically create it just use the following configuration:

[UserProfilesService]
    LocalServiceModule =
"OpenSim.Services.UserProfilesService.dll:UserProfilesService"
    Enabled = true
    UserAccountService =
OpenSim.Services.UserAccountService.dll:UserAccountService
    AuthenticationServiceModule =
"OpenSim.Services.AuthenticationService.dll:PasswordAuthenticationService"
